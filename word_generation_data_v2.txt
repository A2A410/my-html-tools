SURPRISE ALLOWANCE:
  Controls how often low-probability choices are forced
  
  Range: 0% (never) → 50% (half the time)
  
  At 20% surprise:
    - Every letter has 20% chance to be selected from bottom 30% of candidates
    - Creates intentional "mistakes" that sound interesting
    - Prevents clustering around safe patterns
  
  Implementation:
    IF random() < surprise_allowance:
        candidates = bottom_30_percent(weighted_candidates)
    ELSE:
        candidates = all_candidates
    END IF

ENTROPY BUDGETS (Per Word):
  Tracks cumulative surprise across entire word
  
  LOW_ENTROPY:     0.0 - 1.2  (dictionary-like)
  MEDIUM_ENTROPY:  1.2 - 2.8  (creative sweet spot)
  HIGH_ENTROPY:    2.8 - 5.0  (experimental)
  CHAOS_ENTROPY:   5.0+       (alien)
  
  Formula:
    entropy += -log2(selected_probability)
  
  Usage:
    IF current_entropy > budget_max:
        switch_to_safer_choices()
    END IF

Pronounceability:
  - Penalize 3+ consonant clusters: -5 per cluster
  - Penalize 4+ vowel runs: -8 per run
  - Reward alternating VC pattern: +3 per alternation
  - Optimal vowel ratio: 35-45% (+10 if in range)
  
  Formula:
    base_score = 50
    score += alternation_bonus
    score -= cluster_penalty
    score += vowel_ratio_bonus
    CLAMP(score, 0, 100)

Syllable Rhythm:
  - Detect syllable breaks (vowel groups)
  - Measure stress pattern (alternating strong/weak)
  - Regular patterns (iambic, trochaic): +20
  - Chaotic patterns: -15
  
  Example:
    "BANANA" → ba-NA-na (trochaic-iambic) → 75/100
    "STRENGTH" → strength (monosyllabic cluster) → 40/100

Transition Smoothness:
  Harsh jumps: P→K, T→Q, B→X, K→Z, etc.
  Smooth jumps: L→M, N→T, R→S, S→T, etc.
  
  harsh_score = (harsh_transitions / total_transitions) × 100
  smoothness_score = 100 - harsh_score

Mouth Comfort:
  Articulation difficulty by letter sequence
  
  Easy: L,M,N,R,S,T + vowels
  Medium: B,D,G,P,K,F,V
  Hard: Z,X,Q,TH,CH,PH clusters
  
  avg_difficulty = sum(letter_difficulty) / length
  comfort_score = 100 - (avg_difficulty × 20)

  - "OR" combination: +4

FUNCTION generate_word_v2(
    max_length = 12,
    temperature = 0.5,
    diversity_mode = STANDARD,
    surprise_allowance = 0.15,
    phonetic_mode = ENGLISH,
    trend_template = NONE,
    context = NONE
):
    // Initialization
    word = ""
    consecutive_count = {}
    entropy_budget = 0.0
    identity = NONE
    path_momentum = NONE
    
    // Apply trend template if specified
    IF trend_template != NONE:
        word = apply_trend_prefix(trend_template)
        max_length = calculate_target_length(trend_template)
    END IF
    
    FOR position FROM 1 TO max_length:
        candidates = []
        
        // STEP 1: Base Probabilities
        base_probs = get_position_favorability(position, phonetic_mode)
        
        FOR each letter IN alphabet:
            prob = base_probs[letter]
            
            // STEP 2: Temperature Adjustment
            prob = prob ^ (1 / temperature)
            
            // STEP 3: Diversity Mode
            prob = apply_diversity_mode(prob, letter, diversity_mode)
            
            // STEP 4: Consecutive Constraint
            IF consecutive_count[letter] >= 3:
                prob = 0
            END IF
            
            // STEP 5: Bigram Filtering
            IF position > 1:
                bigram_prob = get_bigram_probability(word[-1], letter, phonetic_mode)
                prob *= bigram_prob
            END IF
            
            // STEP 6: Trigram Bonus
            IF position > 2:
                IF is_common_trigram(word[-2:], letter, phonetic_mode):
                    prob *= 1.5
                END IF
            END IF
            
            // STEP 7: Path Memory Bias
            IF position >= 4:
                IF identity == NONE:
                    identity = detect_identity(word)
                END IF
                
                IF identity != NONE:
                    IF matches_identity(letter, identity):
                        prob *= 1.3  // Identity coherence bonus
                    END IF
                END IF
                
                IF path_momentum == NONE:
                    path_momentum = detect_momentum(word)
                END IF
                
                IF path_momentum != NONE:
                    IF matches_momentum(letter, path_momentum):
                        prob *= 1.2  // Flow momentum bonus
                    END IF
                END IF
            END IF
            
            // STEP 8: Phonotactic Validation
            IF NOT is_phonotactically_valid(word, letter, position, phonetic_mode):
                prob = 0
            END IF
            
            // STEP 9: Entropy Budget Check
            letter_entropy = -log2(prob)
            IF (entropy_budget + letter_entropy) > get_max_entropy(temperature):
                prob *= 0.5  // Reduce probability if over budget
            END IF
            
            candidates.append({letter: prob})
        END FOR
        
        // STEP 10: Normalize
        total = sum(candidates.values())
        FOR each candidate:
            candidate.prob /= total
        END FOR
        
        // STEP 11: Surprise Injection
        IF random() < surprise_allowance:
            candidates = filter_bottom_30_percent(candidates)
        END IF
        
        // STEP 12: Selection
        selected = weighted_random_choice(candidates)
        word += selected
        
        // STEP 13: Update State
        IF word[-1] == selected:
            consecutive_count[selected] += 1
        ELSE:
            consecutive_count = {selected: 1}
        END IF
        
        selected_prob = get_probability(selected, candidates)
        entropy_budget += -log2(selected_prob)
        
        // STEP 14: Termination Check
        IF should_terminate_v2(word, position, identity, context):
            BREAK
        END IF
    END FOR
    
    // STEP 15: Apply Trend Suffix
    IF trend_template != NONE:
        word = apply_trend_suffix(word, trend_template)
    END IF
    
    // STEP 16: Aesthetic Evaluation
    scores = {
        phonetic_flow: calculate_phonetic_flow(word),
        visual_balance: calculate_visual_balance(word),
        sentence_fit: calculate_sentence_fit(word, context),
        composite: calculate_composite_score(word, context)
    }
    
    RETURN {
        word: word,
        scores: scores,
        identity: identity,
        entropy: entropy_budget,
        metadata: {
            temperature: temperature,
            mode: phonetic_mode,
            trend: trend_template
        }
    }
END FUNCTION

BATCH GENERATION (Word Hunting):
  Generate multiple candidates and rank by composite score
  
  FUNCTION hunt_words(
      count = 10,
      filters = [],
      sort_by = "composite"
  ):
      candidates = []
      
      FOR i FROM 1 TO count:
          word_result = generate_word_v2(...)
          
          // Apply filters
          passes = TRUE
          FOR each filter:
              IF NOT filter.test(word_result):
                  passes = FALSE
                  BREAK
              END IF
          END FOR
          
          IF passes:
              candidates.append(word_result)
          END IF
      END FOR
      
      // Sort by specified score
      candidates.sort(key = lambda x: x.scores[sort_by], reverse = TRUE)
      
      RETURN candidates
  END FUNCTION
